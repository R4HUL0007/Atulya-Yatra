# Database Initialization
@app.before_request
def create_tables():
    if not hasattr(app, 'has_run_once'):
        app.has_run_once = True
        db.create_all()
        # Insert sample data if the tables are empty
        if Place.query.count() == 0:
            sample_places()
        if State.query.count() == 0:
            sample_states()
        if Feedback.query.count() == 0:
            sample_feedback()
